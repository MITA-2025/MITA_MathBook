
\Formula{幂函数}{
    $\Forall{\mu}{\bfR}$
    \Equation{
        \braI{x}'=1\quad,\quad\braI{x^{\mu}}'=\mu x^{\mu-1}\ .
    }
}

\Formula{指数函数&对数函数}<指数函数\&对数函数>{
	$\Forall{a}{\braI{0\and+\infty}\setminus\braIII{1}}$
    \Equation{
        \braI{a^{x}}'=a^{x}\ln a\quad,\quad\braI{\log_{\,a\!}x}'=\frac{1}{x\ln a}\ .
    }
}

\Formula{三角函数}{
    \Equation{
        \braI{\sin x}'=\cos x\quad,\quad\braI{\cos x}'=-\sin x\quad,\quad\braI{\tan x}'=\sec^{2}x\ ,\\[-15mm]
    }
    \Equation{
        \braI{\sec x}'=\tan x\sec x\quad,\quad\braI{\csc x}'=-\cot x\csc x\quad,\quad\braI{\cot x}'=-\csc^{2}x\ .
    }
}

\Formula{反三角函数}{
    \Equation{
        \braI{\arcsin x}'=\frac{1}{\sqrt{1-x^{2}}}\quad,\quad\braI{\arccos x}'=-\frac{1}{\sqrt{1-x^{2}}}\quad,\quad\braI{\arctan x}'=\frac{1}{x^{2}+1}\ ,\\[-13mm]
    }
    \Equation{
        \braI{\arcsec x}'=\frac{1}{\braIV{x}\sqrt{x^{2}-1}}\quad,\quad\braI{\arccsc x}'=-\frac{1}{\braIV{x}\sqrt{x^{2}-1}}\quad,\quad\braI{\arccot x}'=-\frac{1}{x^{2}+1}\ .
    }
}

\Formula{双曲函数}{
    \Equation{
        \braI{\sinh x}'=\cosh x\quad,\quad\braI{\cosh x}'=\sinh x\quad,\quad\braI{\tanh x}'=\mathrm{sech}^{2}\,x\ ,\\[-15mm]
    }
    \Equation{
        \braI{\sech x}'=-\tanh x\,\sech x\quad,\quad\braI{\csch x}'=-\coth x\,\csch x\quad,\quad\braI{\coth x}'=-\mathrm{csch}^{2}\,x\ .
    }
}

\Formula{反双曲函数}{
        \Equation{
            \braI{\arsinh x}'=\frac{1}{\sqrt{x^{2}+1}}\quad,\quad\braI{\arcosh x}'=\frac{1}{\sqrt{x^{2}-1}}\quad,\quad\braI{\artanh x}'=\frac{1}{1-x^{2}}\ ,\\[-13mm]
        }
        \Equation{
            \braI{\arsech x}'=-\frac{1}{x\sqrt{1-x^{2}}}\quad,\quad\braI{\arcsch x}'=-\frac{1}{\braIV{x}\sqrt{x^{2}+1}}\quad,\quad\braI{\arcoth x}'=\frac{1}{1-x^{2}}\ .
        }
}
