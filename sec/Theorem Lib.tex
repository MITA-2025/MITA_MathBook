
\Formula{AM\—GM不等式}{
	设 $a_{1}\and\cdots\and a_{n}>0$ , 则
	\Equation{
		\braI{\prod_{k=1}^{n}a_{k}}^{\tfrac{1}{n}}\leqslant\frac{1}{n}\sum_{k=1}^{n}a_{k}\ ,
	}
	其中等号当且仅当 $a_{1}=\cdots=a_{n}$ 时取得.
}

\Formula{Stolz定理}{
	设 $L$ 是广义实数, 数列 $(a_{n})_{n=1}^{\infty}\and(b_{n})_{n=1}^{\infty}$ 满足
	\Equation{
		\lim_{n\to\infty}\frac{a_{n+1}-a_{n}}{b_{n+1}-b_{n}}=L\ ,
	}
	+ 若 $(b_{n})_{n=1}^{\infty}$ 严格递增且 $\lim_{n\to\infty}b_{n}=+\infty$ , 则 $\lim_{n\to\infty}\sfrac{a_{n}}{b_{n}}=L$ ;
	+ 若 $(b_{n})_{n=1}^{\infty}$ 严格递减且 $\lim_{n\to\infty}a_{n}=\lim_{n\to\infty}b_{n}=0$ , 则 $\lim_{n\to\infty}\sfrac{a_{n}}{b_{n}}=L$ .
}

\Formula{Stirling公式}{
	\Equation{
		n\,!\sim\sqrt{2\pi}\cdot\rme^{-n}n^{n+1/2}\ .\quad\braI{n\to\infty}
	}
}

\Formula{Wallis公式}{
	\Equation{
		\prod_{n=1}^{\infty}\frac{4n^{2}}{4n^{2}-1}=\frac{\pi}{2}\ .
	}
}

\Formula{$^{*}$Banach不动点定理}{
	设 $A$ 是 $\bfR$ 的闭子集, 数列 $(a_{n})_{n=1}^{\infty}$ 由递推式 $a_{n+1}=f\braI{a_{n}}$ 给出, 其中 $f:A\to A$ , 若存在实数 $0<q<1$ 使得
	\Equation{
		\Forall{x\and y}{A}
		\braIV{f\braI{x}-f\braI{y}\!}\leqslant q\braIV{x-y}\ ,
	}
	则存在唯一的 $x^{*}\in A$ 使得 $f\braI{x^{*}}=x^{*}$ 并且 $\lim_{n\to\infty}a_{n}=x^{*}$ .
}

\Formula{$^{*}$Euler公式}{
	%$\FORALL{z}{\bfC}$
	\Equation{
		\rme^{\rmi z}=\cos z+\rmi\sin z\ ,
	}
	特别地, 当 $z\in\bfR$ 时有 $\cos z=\Re\braI{\rme^{\rmi z}}$ 和 $\sin z=\Im\braI{\rme^{\rmi z}}$ .
}
