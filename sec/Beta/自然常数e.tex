
\Theorem{自然常数的存在性}($\bm\rme$的存在性){
    \Equation{
        a_{n}=\braI{1+\frac{1}{n}}^{n}
    }
    收敛.
}
\Proof{证明I}{
    \par 设 $a_{n}=\braI{1+\sfrac{1}{n}}^{n}$ , 根据AM\—GM不等式, 对一切正整数 $n$ 我们有
    \Equation{
        a_{n}=\braI{1+\frac{1}{n}}^{n}\times1\leqslant\braI{\frac{n\braI{1+\sfrac{1}{n}}+1}{n+1}}^{n+1}=\braI{1+\frac{1}{n+1}}^{n+1}=a_{n+1}\ ,\\[-14mm]
    }
    \Equation{
        a_{n}=4\braII{\!\braI{1+\frac{1}{n}}^{n}\times\frac{1}{2}\times\frac{1}{2}}\leqslant4\braI{\frac{n\braI{1+\sfrac{1}{n}}+\sfrac{1}{2}+\sfrac{1}{2}}{n+2}}^{n+2}=4\ ,
    }
    于是 $(a_{n})_{n=1}^{\infty}$ 单调递增且存在上界, 由单调有界准则知该极限存在.
}
\Proof{证明II}{
    \par 设 $a_{n}=\braI{1+\sfrac{1}{n}}^{n}$ , 我们有
    \Equation{
        a_{n}=\braI{1+\frac{1}{n}}^{n}=\sum_{k=0}^{n}\rmC_{n}^{k}n^{-k}=\sum_{k=0}^{n}\frac{1}{k\,!}\prod_{m=1}^{k-1}\braI{1-\frac{m}{n}}\ ,
    }
    同理有
    \Equation{
        a_{n+1}=\sum_{k=0}^{n+1}\frac{1}{k\,!}\prod_{m=1}^{k-1}\braI{1-\frac{m}{n+1}}=\frac{1}{\braI{n+1}^{n+1}}+\sum_{k=0}^{n}\frac{1}{k\,!}\prod_{m=1}^{k-1}\braI{1-\frac{m}{n+1}}\ ,
    }
    于是 $(a_{n})_{n=1}^{\infty}$ 单调递增, 并且又有
    \Equation{
        a_{n}\leqslant\sum_{k=0}^{n}\frac{1}{k\,!}\leqslant1+\sum_{k=0}^{n-1}\frac{1}{2^{k}}=3-\frac{1}{2^{n-1}}<3\ ,
    }
    即 $(a_{n})_{n=1}^{\infty}$ 存在上界, 由单调有界准则知该极限存在.
}

\Definition{自然常数}(自然常数 $\bm\rme$){
    我们定义\newconcept{自然常数}为
    \Equation{
        \rme:=\lim_{n\to\infty}\braI{1+\frac{1}{n}}^{n}\approx2.7182\,8182\,8459\,0452\,3536\ .
    }
}

\Theorem{自然常数是无理数}($\bm\rme$是无理数){}
\Proof{证明\method{反证法}}{
    \par 假设存在正整数 $p\and q$ 使得 $\rme=\sfrac{p}{q}$ , 则显然
    \Equation{
        q\,!\braI{\rme-\sum_{k=0}^{q}\frac{1}{k\,!}}
    }
    是正整数, 但我们有
    \Equation{
        q\,!\braI{\rme-\sum_{k=0}^{q}\frac{1}{k\,!}}=\sum_{k=q+1}^{\infty}\frac{q\,!}{k\,!}=\sum_{k=1}^{\infty}\frac{1}{\braI{q+1}\braI{q+2}\cdots\braI{q+k}}<\sum_{k=1}^{\infty}\frac{1}{\braI{q+1}^{k}}=\frac{1}{q}\ ,
    }
    这就产生了矛盾, 因此 $\rme$ 是无理数.
}
